278int grafo_linea_desde_paso(int algoritmo, int paso_actual, int total_pasos) {
279 static const int mapa_bfs[5] = {1, 2, 4, 8, 11};
280 static const int mapa_dfs[5] = {1, 2, 4, 8, 11};
281 static const int mapa_dijkstra[5] = {1, 2, 6, 9, 11};
282 static const int mapa_bf[5] = {1, 2, 5, 6, 10};
283 static const int mapa_prim[5] = {1, 2, 4, 5, 7};
284 static const int mapa_kruskal[5] = {1, 2, 4, 5, 8};
285 int bucket;
286
287 if (total_pasos <= 1) {
288 bucket = 0;
289 } else {
290 if (paso_actual < 0) {
291 paso_actual = 0;
292 }
293 if (paso_actual >= total_pasos) {
294 paso_actual = total_pasos - 1;
295 }
296 bucket = (paso_actual * 4) / (total_pasos - 1);
297 if (bucket < 0) {
298 bucket = 0;
299 }
300 if (bucket > 4) {
301 bucket = 4;
302 }
303 }
304
305 switch (algoritmo) {
306 case GRAFO_ALGO_BFS:
307 return mapa_bfs[bucket];
308 case GRAFO_ALGO_DFS:
309 return mapa_dfs[bucket];
310 case GRAFO_ALGO_DIJKSTRA:
311 return mapa_dijkstra[bucket];
312 case GRAFO_ALGO_BELLMAN_FORD:
313 return mapa_bf[bucket];
314 case GRAFO_ALGO_PRIM:
315 return mapa_prim[bucket];
316 case GRAFO_ALGO_KRUSKAL:
317 return mapa_kruskal[bucket];
318 default:
319 return 1;
320 }
321}

365 } else if ((dist_prev[i] < 1000000000 && dist[i] > dist_prev[i]) ||

367 metricas.empeora++;
368 } else {
369 metricas.sin_cambio++;
370 }
371 }
372
373 return metricas;
