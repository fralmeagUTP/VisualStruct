36static void grafo_controller_construir_script(GrafoController *controller) {

64 controller->vertice_destino)
65 : grafo_bellman_ford(controller->grafo_tad, controller->vertice_inicio,
66 controller->vertice_destino);

70 if (limite_aristas > 256) {

76 controller->script_aristas_count = limite_aristas;

95 int i;

98 controller->script_aristas[i] = c.aristas[i];
99 }
100 controller->script_aristas_count = limite_aristas;

103 break;
104 }
105 default:
106 break;
107 }

109 if (controller->script_vertices_count > 0) {

111 } else if (controller->script_aristas_count > 0) {
112 controller->total_pasos = controller->script_aristas_count;
113 } else {
114 controller->total_pasos = 1;

118 controller->total_pasos += 1;
119 }
