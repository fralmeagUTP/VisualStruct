14GrafoState grafo_state_init(void) {
15 GrafoState estado = {0};
16 estado.cantidad_vertices = 0;
17 estado.cantidad_aristas = 0;
18 estado.es_dirigido = false;
19 estado.algoritmo_activo = GRAFO_ALGO_NINGUNO;
20 estado.paso_algoritmo = 0;
21 estado.total_pasos = 0;
22
23 for (int i = 0; i < 64; i++) {
24 estado.vertices[i].id = -1;
25 estado.vertices[i].estado = GRAFO_VÉRTICE_NORMAL;
26 estado.vertices[i].visible = false;
27 estado.vertices[i].distancia = 0;
28 estado.vertices[i].predecesor = -1;
29 estado.vertices[i].orden_visitacion = 0;
30 }
31
32 for (int i = 0; i < 256; i++) {
33 estado.aristas[i].origen = -1;

35 estado.aristas[i].estado = GRAFO_ARISTA_NORMAL;
36 estado.aristas[i].visible = false;
37 estado.aristas[i].orden_examinacion = 0;
38 }
39
40 strcpy(estado.mensaje_estado, "Estado inicial");
