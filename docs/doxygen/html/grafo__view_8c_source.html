102void grafo_vista_punto_en_linea(float x1, float y1, float x2, float y2,
103 float distancia, float *px, float *py) {
104 if (!px || !py) return;
105
106 float dx = x2 - x1;
107 float dy = y2 - y1;
108 float len = sqrtf(dx * dx + dy * dy);
109
110 if (len < 0.001f) {
111 *px = x1;
112 *py = y1;
113 return;
114 }
115
116 *px = x1 + (dx / len) * distancia;
117 *py = y1 + (dy / len) * distancia;
118}

120void grafo_vista_dibujar_flecha(float x1, float y1, float x2, float y2,
121 float grosor, Color color) {
122 DrawLineEx((Vector2){x1, y1}, (Vector2){x2, y2}, grosor, color);
123
124 /* Calcular punta de flecha */
125 float dx = x2 - x1;
126 float dy = y2 - y1;
127 float len = sqrtf(dx * dx + dy * dy);
128
129 if (len > 0.001f) {
130 float px = x2 - (dx / len) * 15.0f;
131 float py = y2 - (dy / len) * 15.0f;
132
133 /* Líneas laterales de punta */
134 float px1 = px + (dy / len) * 5.0f;
135 float py1 = py - (dx / len) * 5.0f;
136 float px2 = px - (dy / len) * 5.0f;
137 float py2 = py + (dx / len) * 5.0f;
138
139 DrawLineEx((Vector2){x2, y2}, (Vector2){px1, py1}, grosor, color);
140 DrawLineEx((Vector2){x2, y2}, (Vector2){px2, py2}, grosor, color);
141 }

144int grafo_vista_detectar_vertice(const GrafoVista *vista, Vector2 mouse_pos) {
145 if (!vista || !vista->estado) return -1;
146
147 for (int i = 0; i < vista->estado->cantidad_vertices; i++) {
148 const GrafoVerticeVisual *v = &vista->estado->vertices[i];
149 if (!v->visible) continue;
150
151 float dx = mouse_pos.x - v->x;
152 float dy = mouse_pos.y - v->y;
153 float dist = sqrtf(dx * dx + dy * dy);
154
155 if (dist <= v->radio * 1.2f) {
156 return v->id;
157 }
158 }
159 return -1;
160}

297void grafo_vista_dibujar_estado(const GrafoVista *vista) {
298 if (!vista || !vista->estado) return;
299
300 int y_offset = (int)vista->area_renderizado.y + 10;
301 int legend_y = (int)(vista->area_renderizado.y + vista->area_renderizado.height - 18.0f);
302
303 /* Mensaje de estado */
304 DrawText(vista->estado->mensaje_estado,
305 (int)vista->area_renderizado.x + 10, y_offset, 14, vista->colores.texto_normal);
306
307 /* Información de algoritmo */
308 if (vista->estado->algoritmo_activo != GRAFO_ALGO_NINGUNO) {
309 char algo_str[128];
310 const char *algo_nombre = "";
311
312 switch (vista->estado->algoritmo_activo) {
313 case GRAFO_ALGO_BFS: algo_nombre = "BFS"; break;
314 case GRAFO_ALGO_DFS: algo_nombre = "DFS"; break;
315 case GRAFO_ALGO_DIJKSTRA: algo_nombre = "Dijkstra"; break;
316 case GRAFO_ALGO_BELLMAN_FORD: algo_nombre = "Bellman-Ford"; break;
317 case GRAFO_ALGO_PRIM: algo_nombre = "Prim"; break;
318 case GRAFO_ALGO_KRUSKAL: algo_nombre = "Kruskal"; break;
319 default: break;
320 }
321
322 snprintf(algo_str, sizeof(algo_str), "Algoritmo: %s | Progreso %d/%d",
323 algo_nombre, vista->estado->paso_algoritmo + 1, vista->estado->total_pasos);
324 DrawText(algo_str, (int)vista->area_renderizado.x + 10, y_offset + 20, 12,
325 vista->colores.texto_destacado);
326 }
327
328 DrawRectangleRounded((Rectangle){vista->area_renderizado.x + 10.0f, (float)legend_y, 9.0f, 9.0f},
329 0.25f, 4, Fade(vista->colores.vertice_actual, 0.55f));
330 DrawText("Activo", (int)vista->area_renderizado.x + 24, legend_y - 2, 10,

333 0.25f, 4, Fade(vista->colores.arista_relajada, 0.55f));
334 DrawText("Procesada", (int)vista->area_renderizado.x + 96, legend_y - 2, 10,
335 vista->colores.texto_normal);
336 DrawRectangleRounded((Rectangle){vista->area_renderizado.x + 172.0f, (float)legend_y, 9.0f, 9.0f},
337 0.25f, 4, Fade(vista->colores.arista_camino_minimo, 0.55f));
338 DrawText("Mejora", (int)vista->area_renderizado.x + 186, legend_y - 2, 10,
