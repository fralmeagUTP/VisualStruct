616 state->ultima_operacion_ok = cp_desencolar(&state->cola_prioridad, &valor, &prioridad);
617 if (state->ultima_operacion_ok) {
618 state->ultimo_valor = valor;
619 state->ultima_prioridad = prioridad;
620 trigger_feedback(state, OPERACION_ELIMINAR);
621 snprintf(state->mensaje_operacion, sizeof(state->mensaje_operacion),
622 "cp_desencolar() -> (%d, p=%d)", valor, prioridad);
623 } else {
624 set_message(state, "cp_desencolar no disponible: cola vacia");
625 }
626 break;
627 case ESTRUCTURA_LISTA:
628 state->ultima_operacion_ok = lista_eliminar_primero(&state->lista, state->input_valor);
629 if (state->ultima_operacion_ok) {
630 state->ultimo_valor = state->input_valor;
631 trigger_feedback(state, OPERACION_ELIMINAR);
632 snprintf(state->mensaje_operacion, sizeof(state->mensaje_operacion),
633 "lista_eliminar_primero(%d)", state->input_valor);
634 } else {
635 set_message(state, "valor no encontrado en lista");
636 }
637 break;
638 case ESTRUCTURA_LISTA_CIRCULAR:
639 state->ultima_operacion_ok =
640 lcir_eliminar_primero(&state->lista_circular, state->input_valor);
641 if (state->ultima_operacion_ok) {
642 state->ultimo_valor = state->input_valor;
643 trigger_feedback(state, OPERACION_ELIMINAR);
644 snprintf(state->mensaje_operacion, sizeof(state->mensaje_operacion),
645 "lcir_eliminar_primero(%d)", state->input_valor);
646 } else {
647 set_message(state, "valor no encontrado en lista circular");
648 }
649 break;
650 case ESTRUCTURA_SUBLISTA:
651 state->ultima_operacion_ok =
652 sublista_eliminar_padre_primero(&state->sublista, state->input_valor);
653 if (state->ultima_operacion_ok) {
654 state->ultimo_valor = state->input_valor;
655 if (state->sublista_padre_activo_ok &&
656 state->sublista_padre_activo == state->input_valor) {
657 state->sublista_padre_activo_ok = false;
658 }
659 trigger_feedback(state, OPERACION_ELIMINAR);
660 snprintf(state->mensaje_operacion, sizeof(state->mensaje_operacion),
661 "sublista_eliminar_padre_primero(%d)", state->input_valor);
662 } else {
663 set_message(state, "padre no encontrado en sublista");
664 }
665 break;
666 case ESTRUCTURA_GRAFO: {
667 state->ultima_operacion_ok = grafo_controller_eliminar_vertice(
668 &state->grafo_controller_state, state->input_valor);
669 if (state->ultima_operacion_ok) {
670 state->ultimo_valor = state->input_valor;
671 trigger_feedback(state, OPERACION_ELIMINAR);
672 snprintf(state->mensaje_operacion, sizeof(state->mensaje_operacion),
673 "grafo_eliminar_vertice(%d)", state->input_valor);
674 } else {
675 snprintf(state->mensaje_operacion, sizeof(state->mensaje_operacion),
676 "Error grafo: %s",
677 grafo_controller_obtener_error(&state->grafo_controller_state));
678 }
679 break;
680 }
681 default:
682 break;
683 }

749void app_state_operacion_grafo_ejecutar_algoritmo(AppState *state, int algoritmo, int inicio,
750 int destino) {
751 GrafoRecorrido recorrido;
752 GrafoCamino camino;
753
754 if (state == NULL || state->estructura_activa != ESTRUCTURA_GRAFO) {
755 return;
756 }
757
758 if (grafo_orden(state->grafo) == 0) {
759 state->ultima_operacion_ok = false;
760 set_message(state, "Error algoritmo: el grafo esta vacio");
761 return;
762 }
763 if ((algoritmo == GRAFO_ALGO_BFS || algoritmo == GRAFO_ALGO_DFS || algoritmo == GRAFO_ALGO_DIJKSTRA ||
764 algoritmo == GRAFO_ALGO_BELLMAN_FORD || algoritmo == GRAFO_ALGO_PRIM) && inicio < 0) {
765 state->ultima_operacion_ok = false;
766 set_message(state, "Error algoritmo: define un vertice de inicio valido");
767 return;
768 }
769 if ((algoritmo == GRAFO_ALGO_DIJKSTRA || algoritmo == GRAFO_ALGO_BELLMAN_FORD) && destino < 0) {
770 state->ultima_operacion_ok = false;
771 set_message(state, "Error algoritmo: define un vertice destino valido");
772 return;
773 }
774
775 state->grafo_algoritmo_seleccionado = algoritmo;
776 state->grafo_vertice_inicio = inicio;
777 state->grafo_vertice_destino = destino;
778 state->operacion_actual = OPERACION_BUSCAR;
779 state->operacion_serial++;
780
781 grafo_controller_seleccionar_algoritmo(&state->grafo_controller_state, algoritmo, inicio,
782 destino);
783 grafo_controller_iniciar_algoritmo(&state->grafo_controller_state);
784
785 state->ultima_operacion_ok = true;
786 switch (algoritmo) {
787 case GRAFO_ALGO_BFS:
788 recorrido = grafo_bfs(state->grafo, inicio);
789 state->ultima_operacion_ok = (recorrido.estado == GRAFO_OK);
790 snprintf(state->mensaje_operacion, sizeof(state->mensaje_operacion),
791 "BFS(inicio=%d) -> %d vertice(s)", inicio, (int)recorrido.cantidad);
792 grafo_liberar_recorrido(&recorrido);
793 break;
794 case GRAFO_ALGO_DFS:
795 recorrido = grafo_dfs(state->grafo, inicio);
796 state->ultima_operacion_ok = (recorrido.estado == GRAFO_OK);
797 snprintf(state->mensaje_operacion, sizeof(state->mensaje_operacion),
798 "DFS(inicio=%d) -> %d vertice(s)", inicio, (int)recorrido.cantidad);
799 grafo_liberar_recorrido(&recorrido);
800 break;
801 case GRAFO_ALGO_DIJKSTRA:
802 camino = grafo_dijkstra(state->grafo, inicio, destino);
803 state->ultima_operacion_ok = (camino.estado == GRAFO_OK);
804 if (camino.estado == GRAFO_ERROR_PESO_NEGATIVO) {
805 snprintf(state->mensaje_operacion, sizeof(state->mensaje_operacion),
806 "Dijkstra no admite pesos negativos");
807 } else if (camino.existe) {
808 snprintf(state->mensaje_operacion, sizeof(state->mensaje_operacion),
809 "Dijkstra(%d->%d) costo=%d", inicio, destino, camino.costo_total);
810 } else {
811 snprintf(state->mensaje_operacion, sizeof(state->mensaje_operacion),
812 "Dijkstra(%d->%d) sin camino", inicio, destino);
813 }
814 grafo_liberar_camino(&camino);
815 break;
816 case GRAFO_ALGO_BELLMAN_FORD:
817 camino = grafo_bellman_ford(state->grafo, inicio, destino);
818 state->ultima_operacion_ok = (camino.estado == GRAFO_OK);
819 if (camino.estado == GRAFO_ERROR_CICLO_NEGATIVO) {
820 snprintf(state->mensaje_operacion, sizeof(state->mensaje_operacion),
821 "Bellman-Ford detecto ciclo negativo");
822 } else if (camino.existe) {
823 snprintf(state->mensaje_operacion, sizeof(state->mensaje_operacion),
824 "Bellman-Ford(%d->%d) costo=%d", inicio, destino, camino.costo_total);
825 } else {
826 snprintf(state->mensaje_operacion, sizeof(state->mensaje_operacion),
827 "Bellman-Ford(%d->%d) sin camino", inicio, destino);
828 }
829 grafo_liberar_camino(&camino);
830 break;
831 case GRAFO_ALGO_PRIM:
832 camino = grafo_prim(state->grafo, inicio);
833 state->ultima_operacion_ok = (camino.estado == GRAFO_OK);
834 snprintf(state->mensaje_operacion, sizeof(state->mensaje_operacion), "Prim(inicio=%d)",
835 inicio);
836 grafo_liberar_camino(&camino);
837 break;
838 case GRAFO_ALGO_KRUSKAL:
839 camino = grafo_kruskal(state->grafo);
840 state->ultima_operacion_ok = (camino.estado == GRAFO_OK);
841 snprintf(state->mensaje_operacion, sizeof(state->mensaje_operacion), "Kruskal() aristas=%d",
842 (int)camino.cantidad);
843 grafo_liberar_camino(&camino);
844 break;
845 default:
846 state->ultima_operacion_ok = false;
847 set_message(state, "Algoritmo no soportado");
848 break;
849 }
850
851 if (!state->ultima_operacion_ok && strncmp(state->mensaje_operacion, "Bellman-Ford detecto ciclo negativo", 36) != 0 &&
852 strncmp(state->mensaje_operacion, "Dijkstra no admite pesos negativos", 34) != 0) {
853 snprintf(state->mensaje_operacion, sizeof(state->mensaje_operacion), "Error algoritmo");
854 } else {
855 trigger_feedback(state, OPERACION_BUSCAR);
856 }
857}
