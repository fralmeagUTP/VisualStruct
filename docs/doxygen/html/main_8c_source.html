63static void append_code_history(char *history, size_t capacity, int *entries, const AppState *app,
64 const char *snippet) {
65 char block[3072];
66 size_t used;
67 int written;
68
69 if (history == NULL || entries == NULL || app == NULL || capacity == 0) {
70 return;
71 }
72
73 if (snippet == NULL || snippet[0] == '\0') {
74 snippet = "// Snippet no disponible";
75 }
76
77 written = snprintf(block, sizeof(block),
78 "/* Paso %d | %s | %s | %s */\n"
79 "// %s\n"
80 "%s\n\n",
81 *entries + 1, estructura_nombre(app->estructura_activa),
82 operacion_nombre(app->operacion_actual),
83 app->ultima_operacion_ok ? "OK" : "Error", app->mensaje_operacion, snippet);
84 if (written < 0) {
85 return;
86 }
87
88 used = strlen(history);
89 if (used + (size_t)written + 1 >= capacity) {
90 snprintf(history, capacity, "/* Historial reiniciado: limite alcanzado */\n\n");
91 used = strlen(history);
92 }
93
94 if (used < capacity - 1) {
95 snprintf(history + used, capacity - used, "%s", block);
96 (*entries)++;
97 }

1047static void edit_active_input(char *buffer, size_t size) {

1050

1053 if (len + 1 < size) {
1054 buffer[len] = (char)key;

1061

1063 buffer[len - 1] = '\0';
1064 }
