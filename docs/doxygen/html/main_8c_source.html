79static void append_code_history(char *history, size_t capacity, int *entries, const AppState *app,
80 const char *snippet) {
81 char block[3072];
82 size_t used;
83 int written;
84
85 if (history == NULL || entries == NULL || app == NULL || capacity == 0) {
86 return;
87 }
88
89 if (snippet == NULL || snippet[0] == '\0') {
90 snippet = "// Snippet no disponible";
91 }
92
93 written = snprintf(block, sizeof(block),
94 "/* Paso %d | %s | %s | %s */\n"
95 "// %s\n"
96 "%s\n\n",
97 *entries + 1, estructura_nombre(app->estructura_activa),
98 operacion_nombre(app->operacion_actual),
99 app->ultima_operacion_ok ? "OK" : "Error", app->mensaje_operacion, snippet);
100 if (written < 0) {
101 return;
102 }
103
104 used = strlen(history);
105 if (used + (size_t)written + 1 >= capacity) {
106 snprintf(history, capacity, "/* Historial reiniciado: limite alcanzado */\n\n");
107 used = strlen(history);
108 }
109
110 if (used < capacity - 1) {
111 snprintf(history + used, capacity - used, "%s", block);
112 (*entries)++;
113 }
114}

897static void draw_scrollable_multiline_text(const char *text, Rectangle viewport, int font_size,
898 Color color, float offset_y) {
899 const char *start = text;
900 const char *line_end;
901 char line[256];
902 int line_index = 0;
903 int line_height = font_size + 6;
904 size_t len;
905
906 if (text == NULL) {
907 return;
908 }
909
910 BeginScissorMode((int)viewport.x, (int)viewport.y, (int)viewport.width, (int)viewport.height);
911 while (*start != '\0') {
912 int y;
913
914 line_end = strchr(start, '\n');
915 len = (line_end == NULL) ? strlen(start) : (size_t)(line_end - start);
916 if (len >= sizeof(line)) {
917 len = sizeof(line) - 1;
918 }
919 memcpy(line, start, len);
920 line[len] = '\0';
921
922 y = (int)(viewport.y - offset_y) + line_index * line_height;
923 if (y + line_height >= (int)viewport.y - line_height && y <= (int)(viewport.y + viewport.height)) {
924 ui_draw_text(line, viewport.x, (float)y, (float)font_size, 0.25f, color, false);
925 }
926
927 line_index++;
928 if (line_end == NULL) {
929 break;
930 }
931 start = line_end + 1;
932 }
933 EndScissorMode();
934}
