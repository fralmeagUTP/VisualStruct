46float grafo_layout_calcular_radio_maximo(const GrafoLayoutConfiguracion *config,
47 int cantidad_vertices) {
48 if (!config || cantidad_vertices <= 0) return 50.0f;
49
50 float ancho_util = config->ancho_panel * (1.0f - 2.0f * config->margen_borde);
51 float alto_util = config->alto_panel * (1.0f - 2.0f * config->margen_borde);
52
53 /* Radio máximo es el menor de ancho/alto dividido entre 2 */
54 float radio_max = (ancho_util < alto_util ? ancho_util : alto_util) / 2.0f;
55
56 /* Ajustar si hay muchos vértices (ángulo inter-vértice muy pequeño) */
57 if (cantidad_vertices > 30) {
58 radio_max *= 0.8f; /* Reducir un 20% para evitar solapamientos */
59 }
60
61 return radio_max;
62}

68bool grafo_layout_calcular_circular(GrafoState *estado, const GrafoLayoutConfiguracion *config) {
69 if (!estado || !config) return false;
70 if (estado->cantidad_vertices == 0) return true;
71
72 float centro_x, centro_y;
73 grafo_layout_obtener_centro(config, &centro_x, &centro_y);
74
75 float radio = grafo_layout_calcular_radio_maximo(config, estado->cantidad_vertices);
76
77 /* Distribuir vértices uniformemente en círculo */
78 for (int i = 0; i < estado->cantidad_vertices; i++) {
79 float angulo = 2.0f * M_PI * i / estado->cantidad_vertices;
80
81 estado->vertices[i].x = centro_x + radio * cosf(angulo);
82 estado->vertices[i].y = centro_y + radio * sinf(angulo);
83 estado->vertices[i].radio = config->radio_vertice;
84 estado->vertices[i].visible = true;
85 }
86
87 /* Marcar aristas como visibles */
88 for (int i = 0; i < estado->cantidad_aristas; i++) {
89 estado->aristas[i].visible = true;
90 }
91
92 return true;
93}

95bool grafo_layout_evitar_colisiones(GrafoState *estado, int idx_vertice, float distancia_minima) {
96 if (!estado || idx_vertice < 0 || idx_vertice >= estado->cantidad_vertices) {
97 return false;
98 }
99
100 GrafoVerticeVisual *v = &estado->vertices[idx_vertice];
101
102 /* Verificar colisiones con otros vértices */
103 bool hay_colision = true;
104 int iteraciones = 0;
105
106 while (hay_colision && iteraciones < 10) {
107 hay_colision = false;
108
109 for (int i = 0; i < estado->cantidad_vertices; i++) {
110 if (i == idx_vertice) continue;
111
112 float dist = grafo_layout_distancia(v->x, v->y,
113 estado->vertices[i].x, estado->vertices[i].y);
114
115 if (dist < distancia_minima) {
116 hay_colision = true;
117
118 /* Empujar hacia afuera */
119 float dx = v->x - estado->vertices[i].x;
120 float dy = v->y - estado->vertices[i].y;
121 float len = sqrtf(dx * dx + dy * dy);
122
123 if (len > 0.001f) {
124 dx /= len;
125 dy /= len;
126 v->x = estado->vertices[i].x + dx * distancia_minima;
127 v->y = estado->vertices[i].y + dy * distancia_minima;
128 }
129 }
130 }
131 iteraciones++;
132 }
133
134 return !hay_colision;
135}
